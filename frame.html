### Change the language using i18n/moment.js ```language-HTML ``` ```js // In the Angular.js controller, specify the interface translations, either literally or with moment angular.module('yourApp', ['angularPikaday']) .config(['pikadayProvider', function(pikadayProvider) { moment.locale("de"); // config moment location, defaults to 'en' or most recently loaded moment.js i18n file pikadayProvider.setConfig({ i18n: { previousMonth : 'Vorheriger Monat', nextMonth : 'Nächster Monat', months : moment.localeData()._months, // or literally ["Januar", "Februar", "März", "April", "Mai", "Juni", "Juli", "August", "September", "Oktober", "November", "Dezember"] weekdays : moment.localeData()._weekdays, weekdaysShort : moment.localeData()._weekdaysShort } }); }]) ```
